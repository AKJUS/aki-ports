IPv4
====

rulenr, subrulenr, anchorname, "0" (unused), interface, reason, action, dir,
version, tos, ecn, ttl, id, offset, flags, protonum (!), protoname (!),
length, src, dst

IPv6
====

rulenr, subrulenr, anchorname, "0" (unused), interface, reason, action, dir,
version, class, flowlabel, hoplimit, protoname (!), protonum (!),
length, src, dst [, HBHOPT ][, DSTOPT ][, FRAG6 ][, ROUTING]

HBHOPT
======

"HBH", PAD1, PADN | PADNTRUNC
       RTALERT,data | RTALERTTRUNC | RTALERTINVALID, len
       JUMBO,data | JUMBOTRUNC | JUMBOINVALID, len
       HOMEADDR,addr, otheroptions? | HOMEADDRESSINVALID, len | HOMEADDRESSTRUNC

DSTOPT
======

"DSTOPT", opts...

FRAG6
=====

"FRAG6", frag

ROUTING
=======

"ROUTING", len, type, segleft, reserved, TRUNC | hdridx, addr

MOBILITY
========

"MOBILITY" [, "BINDINGACK", status, flags, sequence, lifetime ]
           [, "BINDINGERR", status, homeaddr ]
           [, "BINDINGUPDATE", seq, flags, lifetime ]
           [, "BRR" (binding request) ]
           [, "HoTI" | "CoTI" (test init), "Home" | "Care-of", cookie ]
           [, "HoT" | "CoT" (test), nonce, "Home" | "Care-of", cookie, keygen-token ]
           [, "TYPE", type , length ]

UDP
===

[IPv4 | IPv6], srcport, dstport, datalen [, label]

TCP
===

[IPv4 | IPv6], srcport, dstport, datalen, flags, seq, ack, window, urg, options [, label]

CARP
====

[IPv4], type, ttl, vhid, version, advskew, advbase [, label]
